http://api.openweathermap.org/data/2.5/forecast/city?id=12345&APPID={APIKEY}

weather

2 homepageId			int(11)		
3 cityId			int(11)		
4 data		text		utf8_general_ci		
5 date		timestamp	on update CURRENT_TIMESTAMP
